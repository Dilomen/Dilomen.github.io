# 声明和断言

声明 TS 依旧会走类型检查，所以优先选择声明

let a: Type = xxx;

let a: number = 1; // 这种都不需要类型声明，直接走类型推断即可
let a: number[] = [1, 2, 3, 4];
let a: [number, string] = [1, '2'];
1

5

断言是告诉 TS 你确定这一定是什么类型，不用走走类型检查

let a = xxx as Type;

let a = 1 as number;
let a = [1, 2, 3, 4] as number[];
let a = [1, '2'] as [number, string];

  console.log('f() evaluated');
  return function() {
    console.log('f(): called');

  console.log('g() evaluated');
  return function() {
    console.log('g(): called');
  };
}
class C {
  @f()
  @g()
  method() {}
}

// 类似洋葱模型: f() evaluated => g() evaluated => g(): called => f(): called

# 类装饰器

对类的构造函数进行更改

@sealed
class Greeter {
  greeting: string;
  constructor(message: string) {
    this.greeting = message;
  }
  greet() {
    return 'Hello, ' + this.greeting;
  }
}
// 装饰类的构造函数为其唯一参数
function sealed(constructor: Function) {
  Object.seal(constructor);
  Object.seal(constructor.prototype);
}

# 方法装饰器

class Greeter {
  greeting: string;
  constructor(message: string) {
    this.greeting = message;
  }

  @enumerable(false)
  greet() {
    return 'Hello, ' + this.greeting;
  }
}

function enumerable(value: boolean) {
  /**
   * target: 静态成员的类的构造函数或实例成员的类的原型
   * propertyKey: 成员的名称 => string 方法名
   * descriptor: 成员的属性描述符
   */
  return function(
    target: any,
    propertyKey: string,
    descriptor: PropertyDescriptor
  ) {
    descriptor.enumerable = value;
  };
}

# class

# 类的修饰符

  • public:公有
    • 在类里面，子类，类外面都可以访问
  • protected:保护类型
    • 在类里面，子类里面可以访问，在类外面没法访问
  • private:私有
    • 在类里面可以访问，子类，类外部度不能访问

# abstract 抽象类

  • 抽象类下声明的抽象方法无须在本类实现
  • 抽象类里也可以写和普通类一样的方法，变量，构造函数等
abstract class Animal {
  abstract call(): any;
  eat() {
    console.log('1');
  }
}

# 类型设计

对于状态要进行区分，不能进行共用类型，容易混乱

// ❌不推荐
interface Response {
  type: 'success' | 'fail';
  data?: IResponseData;
  error?: IResponseError;
}
// ✅推荐
interface successResponse {
  type: 'success';
  data: IResponseData;
}
interface failResponse {
  type: 'fail';
  error: IResponseError;
}

用类型来代替注释 使用更符合场景命名方式

优先使用接口的联合，而不是联合的接口

// ❌不推荐，联合的接口存在属性的交叉，导致混乱 如FillLayout和LinePaint也能成立
interface Layer {
  layout: FillLayout | LineLayout | PointLayout;
  paint: FillPaint | LinePaint | PointPaint;
}
// ✅推荐，接口的联合更精确
interface FillLayer {
  layout: FillLayout;
  paint: FillPaint;
}
interface LineLayer {
  layout: LineLayout;
  paint: LinePaint;
}
interface PointLayer {
  layout: PointLayout;
  paint: PointPaint;
}
type Layer = FillLayer | LineLayer | PointLayer;

# 使用 any

将 any 类型使用最窄的范围

// ❌不推荐，这样所有引用a值的类型都是any
let a: any = fn();
fn1(a);

// ✅推荐，只在fn1调用时，a的类型为any，对其余地方没有影响
let a = fn();
fn1(a as any);
