# 概述

泛型就是对类型的逻辑编码，可以通过其灵活性来拓展应对运行时的动态类型的变化。

# 使用

# 类型编程

类型挑战 (opens new window)

在类型系统里实现一个类似 Array.push 的Push<T, U> 方法 (opens new window)

type Push<T extends unknown[], U> = [...T, U];
Push<[1], 1>
// --------------------------------------------------
[1, 1]
1
2
3
4

实现一个通用MyReadonly2<T, K> (opens new window)，它带有两种类型的参数 T 和 K。 K 指定应设置为 Readonly 的 T 的属性集。如果未提供 K，则应使所有属性都变为只读，就像普通的 Readonly<T>一样

type MyReadonly2<T, K extends keyof T = keyof T> = { readonly [P in K]: T[P] } & Omit<T, K>;

interface Todo1 {
  title: string;
  description?: string;
  completed: boolean;
}

MyReadonly2<Todo1, 'title' | 'description'>
// --------------------------------------------------
interface Todo1 {
  readonly title: string;
  readonly description?: string;
  completed: boolean;
}

# -- Partial<Type>

interface IPerson {
  name: string;
  age: number;
}
/**
 * Partial<IPerson> {
 *  name?: string | undefined;
 *  age?: number | undefined;
 * }
 */
const person: Partial<IPerson> = {};

# -- Readonly<Type>

将所有属性变成只读，不可修改

interface IPerson {
  name: string;
  age: number;
}
/**
 * Readonly<IPerson> {
 *  readonly name: string;
 *  readonly age: number;
 * }
 */
const person: Readonly<IPerson> = {
  name: 'Dilomen',
  age: 26,
};

person.name = 'Alice'; // Error: it is a read-only property.

# -- Record<Keys, Type>

创建一个对象类型

/**
 * Record<'name', string> {
 *  name: string;
 * }
 */
const person: Record<'name', string> = { name: 'Dilomen' };

interface IPersonInfo {
  name: string;
  age: number;
}
type Names = 'Dilome' | 'Alice';
/**
 * Record<Names, IPersonInfo> {
 *  Dilomen: IPersonInfo;
 *  Alice: IPersonInfo;
 * }
 */
const students: Record<Names, IPersonInfo> = {
  name: 'Dilomen',
  age: 26,
};

# -- Exclude<UnionType, ExcludedMembers>

排除联合类型中的某些类型

type IUnionType = 'a' | 'b' | 'c' | 'd';
/**
 * Exclude<IUnionType, 'b'> = 'a' | 'c' | 'd';
 */
const typeExcludedB: Exclude<IUnionType, 'b'> = 'a';

# -- ThisType<Type>

用作于上下文 this 标记，需在 tsconfig.json 配置 noImplicitThis 属性才能开启

const foo: { bar: any } & ThisType<{ a: number }> = {
  bar() {
    console.log(this.a); // ok
    console.log(this.b); // Error: 类型“{ a: number; }”上不存在属性“b”
  },
};
