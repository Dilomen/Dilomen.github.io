题目： 检查 npm 依赖项中有没有循环依赖 例如：a->b b->a， a->b->c c->a， （不用考虑版本，只考虑包名即可） 入参 pkgs 为项目中所有的依赖（包括子依赖） 返回 true/false

pkg 数据结构为 package.json, 如有三个包 a, b, c,入参为

[
  {
    name: "a",
    dependencies: {
      b: "^1.0.0",
      d: "^1.0.0",
    },
  },
  {
    name: "b",
    dependencies: {
      c: "^1.0.0",
    },
  },
  {
    name: "c",
    dependencies: {
      a: "^1.0.0",
    },
  },
];

思路：dfs，深度优先搜索算法，将统计整个链路，一旦 key 在链路中存在了，就表示循环依赖了

function create(state) {
  let initState = {};
  for (let i = 0; i < state.length; i++) {
    const { name, dependencies } = state[i];
    initState[name] = {
      ...dependencies,
    };
  }

  Object.keys(initState).forEach((key) => {
    handle(initState[key], [key], initState);
  });
  return initState;
}

function handle(state, path = [], initState) {
  if (!state) return;
  Object.keys(state).forEach((key) => {
    let newPath = path.concat(key);
    if (path.includes(key)) {
      throw new Error("循环依赖了");
    }
    if (initState.hasOwnProperty(key)) {
      state[key] = {
        val: {
          ...initState[key],
        },
        path: newPath,
      };
    }
    handle(state[key].val, newPath, initState);
  });
}

const data = [
  {
    name: "a",
    dependencies: {
      b: "^1.0.0",
      d: "^1.0.0",
    },
  },
  {
    name: "b",
    dependencies: {
      c: "^1.0.0",
    },
  },
  {
    name: "c",
    dependencies: {
      l: "^1.0.0",
    },
  },
];

create(data);
