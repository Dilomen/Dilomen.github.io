## 概述

遍历就是通过一条线路对一个集合从头到尾进行读取（访问）

## 树(或者图)

### 深度优先遍历

递归版

```js
function deepTraversal(node) {
  if (!node) return;
  let deepNodes = [];
  deepNodes.push(node);
  if (node.children && node.children.length > 0) {
    for (let i = 0; i < node.children; i++) {
      deepTraversal(node.children[i]);
    }
  }
  return deepNodes;
}
```

非递归版

```js
function deepTraversal(node) {
  if (!node) return;
  let deepNodes = [];
  let task = [node];
  while (task.length !== 0) {
    let item = task.pop();
    deepNodes.push(item);
    if (item.children && item.children.length > 0) {
      for (let i = 0; i < item.children; i++) {
        task.push(node.children[i]);
      }
    }
  }
  return deepNodes;
}
```

### 广度优先遍历

递归版

```js
function wideTraversal(node) {
  if (!nodes) return;
  let wideNodes = [];
  let i = 0;
  wideNodes.push(node);
  wideTraversal(node.nextNode);
  node = wideNodes[i];
  ++i;
  wideTraversal(node.firstNode);
  return deepNodes;
}
```

非递归版

```js
function wideTraversal(node) {
  if (!nodes) return;
  let wideNodes = [];
  let i = 0;
  while (node !== null) {
    wideNodes.push(node);
    node = wideNodes[i];
    ++i;
    if (node.children && node.children.length > 0) {
      for (let i = 0; i < node.children; i++) {
        wideNodes.push(node.children[i]);
      }
    }
  }
  return deepNodes;
}
```
