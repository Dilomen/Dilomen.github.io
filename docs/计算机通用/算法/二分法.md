## 二分法

对区间[a,b]的数据不断一分为二，逐渐逼近目标值，最终得到目标近似值的方法

## 基本写法

```js
function binarySearch(arr, target) {
  let low = 0;
  let high = arr.length - 1;
  while (low < high) {
    let mid = parseInt((low + high) / 2);
    if (target > arr[mid]) {
      ++low;
    } else if (target < arr[mid]) {
      --high;
    } else {
      return mid;
    }
  }
  return -1;
}
```

## 再优化

上述其实只做了一次二分，剩下的都是一位索引移步，可以每一次都进行二分

```js
function binarySearch(arr, target) {
  if (!arr || !arr.length) return -1;
  if (arr.length === 1) return arr[0] === target ? 0 : -1;
  let low = 0;
  let high = arr.length - 1;
  while (low < high) {
    let mid = parseInt((low + high) / 2);
    if (target > arr[mid]) {
      low = mid + 1;
    } else if (target < arr[mid]) {
      high = mid - 1;
    // 因为pasreInt是向下取整，所以存在预期值是下一个索引
    } else if ((mid + 1 <= arr.length) && (target === arr[mid + 1])) {
      return mid + 1;
    } else {
      return mid;
    }
  }
  return -1;
}
```
